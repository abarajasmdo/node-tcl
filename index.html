node-tcl

Node.js Tcl bindings

Usage

var tcl = require( 'tcl' );

tcl.cmd( 'info tclversion', function ( err, result ) {
    console.log( result.data() );
} );

tcl.cmd( 'info commands', function ( err, result ) {
    console.log( result.toArray() );
} );
